{
	stu a[N],max,temp;

	for (int i = 0; i < n; i++)
	{
		a[i].num= i + 1;
		cin >> a[i].chi_sco >> a[i].eng_sco >> a[i].mat_sc0;
        a[i].sum = a[i].chi_sco + a[i].eng_sco + a[i].mat_sc0;
	}
	if(--k==0)
		cout << endl;
	for (int i = 0; i < n-1; i++)
	{
		t = i;
		for (j = i+1; j < n; j++)
		{
			if (a[t].sum < a[j].sum)
			{
				t = j;
			}
			else if (a[t].sum == a[j].sum) //总分相同
			{
				if (a[t].chi_sco < a[j].chi_sco)
				{
					t = j;
				}
				else if (a[t].chi_sco == a[j].chi_sco) //语文成绩相同
				{
					if (a[t].num > a[j].num)
					{
						t = j;
					}
				}
			}
		}
		temp = a[i];
		a[i] = a[t];
		a[t] = temp;
	}
	for (int i = 0; i < 5; i++)
	{
		cout << a[i].num << " " << a[i].sum << endl;
	}
	return 0;
}
